ll eular(ll n) {
    ll res = 1;
    for (ll i = 2; i * i <= n; i++)
        if (n % i == 0) {
            n = n / i;
            res *= i - 1;
            while (n % i == 0) {
                n = n / i;
                res *= i;
            }
        }
    if (n > 1) res *= n - 1;
    return res;
}
