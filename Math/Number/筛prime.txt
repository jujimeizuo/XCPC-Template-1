bool isPrime[MAX];
int prime[MAX], num;

void makePrime(int siz) {
    num = 0; memset(isPrime, true, sz(isPrime));
    isPrime[0] = isPrime[1] = num = 0;
    for (int i = 2; i <= siz; i++) {
        if (isPrime[i]) prime[++num] = i;
        for (int j = 1; j <= num && i * prime[j] <= siz; j++) {
            isPrime[i * prime[j]] = 0;
            if (!(i % prime[j]))break;
        }
    }
    return;
}
